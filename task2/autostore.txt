CREATE TABLE autostore_cars (
id INT(6) AUTO_INCREMENT,
make_id INT(6) NOT NULL,
model VARCHAR(255) NOT NULL,
year_model INT(4) NOT NULL,
engine DOUBLE NOT NULL,
speed DOUBLE NOT NULL,
price DOUBLE NOT NULL,
PRIMARY KEY(id),
FOREIGN KEY (brand_id) REFERENCES autoshop_brand(id)
);

CREATE TABLE autostore_make (
id INT(6) AUTO_INCREMENT PRIMARY KEY,
make VARCHAR(255) NOT NULL
);

CREATE TABLE autostore_color (
id INT(6) AUTO_INCREMENT PRIMARY KEY,
color VARCHAR(255) NOT NULL
);

CREATE TABLE autostore_car_color (
id INT(6)  AUTO_INCREMENT PRIMARY KEY,
car_id INT(6) NOT NULL,
color_id INT(6) NOT NULL
);

CREATE TABLE autostore_client_order (
id INT(6)  AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(255) NOT NULL,
surname VARCHAR(255) NOT NULL,
payment ENUM("cash","credit_card") 
);

CREATE TABLE autostore_car_order (
id INT(6)  AUTO_INCREMENT PRIMARY KEY,
car_id INT(6) NOT NULL,
order_id INT(6) NOT NULL
);
